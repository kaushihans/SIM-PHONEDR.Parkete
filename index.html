let currentFirma = null; let data = JSON.parse(localStorage.getItem("db")) || { dpd: { counter: 1000, customers: {} }, gls: { counter: 2000, customers: {} }, hermes: { counter: 3000, customers: {} } }; function save(){ localStorage.setItem("db", JSON.stringify(data)); } /* NAVIGATION */ function openFirma(f){ currentFirma = f; document.getElementById("dashboard").style.display = "none"; document.getElementById("app").style.display = "block"; const names = { dpd:"DPD", gls:"GLS", hermes:"Hermes" }; document.getElementById("header").innerText = names[f]; document.getElementById("header").style.background = getColor(f); render(); } function back(){ document.getElementById("dashboard").style.display = "grid"; document.getElementById("app").style.display = "none"; } /* COLOR */ function getColor(f){ if(f==="dpd") return "var(--dpd)"; if(f==="gls") return "var(--gls)"; if(f==="hermes") return "var(--hermes)"; } /* ADD PACKAGE */ function addPackage(){ let name = document.getElementById("name").value.trim(); let size = document.getElementById("size").value; let form = document.getElementById("form").value; let info = document.getElementById("info").value; if(!name) return; let firma = data[currentFirma]; // 👇 Kunde existiert nicht -> neu erstellen if(!firma.customers[name]){ firma.customers[name] = { id: firma.counter++, packages: [] }; } // Paket hinzufügen firma.customers[name].packages.push({ size, form, info, time: Date.now() }); save(); clearInputs(); render(); } /* CLEAR INPUTS */ function clearInputs(){ document.getElementById("name").value = ""; document.getElementById("info").value = ""; } /* RENDER */ function render(){ let list = document.getElementById("list"); list.innerHTML = ""; let customers = data[currentFirma].customers; for(let name in customers){ let c = customers[name]; let html = `

📦 ID ${c.id} – ${name}

`; c.packages.forEach((p, i)=>{ html += `
Paket ${i+1}
Größe: ${p.size}
Form: ${p.form}
Info: ${p.info || "-"}

`; }); html += `
`; list.innerHTML += html; } }
